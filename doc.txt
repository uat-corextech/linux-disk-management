Disk Management
#command use to list disk and partitions
parted -l

#Checking Disk Space and Usage
df -h

#To see disk usage at the directory level
du -sh /home

#To list all attached storage devices
lsblk 

#list storage device with full path and file system
lsblk -fp

# print partitions
parted /dev/xvdb print

#Create a new partition table (⚠️ this wipes all partitions)
parted /dev/xvdb mklabel msdos
parted /dev/xvdb mklabel gpt

#Creating Partitions
#1MiB ensures alignment (instead of 0%).
#Start/End can be in MB, GB, %, or sector numbers.
#Filesystem type (ext4, xfs, etc.) is only informational

parted /dev/xvdb mkpart primary ext4 1MiB 100MiB

#expand partition:
parted /dev/xvdb resizepart 1 150MiB

#Shrink partition:
parted /dev/xvdb resizepart 1 80MiB

#After partitioning, format file system
mkfs -t ext4 /dev/xvdb1
mkfs -t xfs /dev/xvdb1
mkswap /dev/xvdb1

#Delete a partition:
parted /dev/xvdb rm 1
