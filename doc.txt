Disk Management
#command use to list disk and partitions
parted -l

#Checking Disk Space and Usage
df -h

#To see disk usage‚ÄÇat the directory level
du -sh /home

#To list all attached storage‚ÄÇdevices
lsblk 

#list storage device with full path and file system
lsblk -fp

# print partitions
parted /dev/xvdb print

#Create a new partition table (‚ö†Ô∏è this wipes all partitions)
parted /dev/xvdb mklabel msdos
parted /dev/xvdb mklabel gpt

#Creating Partitions
#1MiB ensures alignment (instead of 0%).
#Start/End can be in MB, GB, %, or sector numbers.
#Filesystem type (ext4, xfs, etc.) is only informational

parted /dev/xvdb mkpart primary ext4 1MiB 100MiB

#expand partition:
parted /dev/xvdb resizepart 1 150MiB

#Shrink partition:
parted /dev/xvdb resizepart 1 80MiB

#After partitioning, format file system
mkfs -t ext4 /dev/xvdb1
mkfs -t xfs /dev/xvdb1
mkswap /dev/xvdb1

#Delete a partition:
parted /dev/xvdb rm 1

#create directory
mkdir -p  /mnt/data

# mount mount point temporary
mount /dev/xvdb1 /mnt/data

#Persistent mount ‚Üí add to /etc/fstab:
UUID=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx /mnt/data ext4 defaults 0 2

#++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++

#Suppose /dev/xvdb1 is 20MiB and you want 50MiB:
sudo umount /dev/xvdb1
sudo parted /dev/xvdb1 resizepart 1 50MiB

#Purpose of e2fsck Before Resizing
#When resizing (especially shrinking) a filesystem like ext4, you must ensure the filesystem is clean and consistent. If there are errors, resizing can:
#Corrupt data
#Lose inodes
#Make the partition unmountable
#That‚Äôs why e2fsck (ext2/ext3/ext4 filesystem checker) is run.

sudo e2fsck -f /dev/sdb1


sudo resize2fs /dev/sdb1

#+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++

#üîπ Growing an XFS Filesystem

sudo parted /dev/sdb resizepart 1 30GiB
sudo xfs_repair /dev/sdb1
sudo umount /dev/sdb1
sudo xfs_repair /dev/sdb1
sudo mount /dev/sdb1 /mnt/data
sudo xfs_growfs /mnt/data
